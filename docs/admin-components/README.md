## 🔧 Component Registration

### Pattern 1: Field Component

Replace the default field UI with a custom component.

```typescript
// src/collections/Exercises/index.ts
export const Exercises: CollectionConfig = {
  slug: 'exercises',
  fields: [
    {
      name: 'content',
      type: 'json',
      required: true,
      admin: {
        description: 'Exercise content with blocks',
        components: {
          Field: '@/components/admin/ExerciseContentEditor#ExerciseContentEditor',
        },
      },
    },
  ],
}
```

**Key Points**:
- ✅ Use path syntax: `'@/path/to/file#ExportName'`
- ✅ Named export: use `#ExportName`
- ✅ Default export: no `#` needed
- ✅ Path relative to project root or `importMap.baseDir`

### Pattern 2: Cell Component

Customize how field values appear in list views.

```typescript
{
  name: 'status',
  type: 'select',
  options: ['draft', 'published'],
  admin: {
    components: {
      Cell: '@/components/admin/StatusCell#StatusCell',
    },
  },
}
```

### Pattern 3: Multiple Component Types

```typescript
{
  name: 'richContent',
  type: 'richText',
  admin: {
    components: {
      Field: '@/components/admin/RichEditor#RichEditor',      // Edit view
      Cell: '@/components/admin/RichCell#RichCell',          // List view
      Label: '@/components/admin/RichLabel#RichLabel',       // Field label
      Description: '@/components/admin/RichDesc#RichDesc',   // Field description
      Error: '@/components/admin/RichError#RichError',       // Error message
    },
  },
```

## 📝 Component Implementation

### Example 1: Basic Custom Field

```typescript
// src/components/admin/MyCustomField/index.tsx
'use client'

import React from 'react'
import { useField } from '@payloadcms/ui'
import type { TextFieldClientComponent } from 'payload'

export const MyCustomField: TextFieldClientComponent = ({ path, field }) => {
  const { value, setValue } = useField<string>({ path })

  return (
    <div>
      <label>{field.label}</label>
      <input
        type="text"
        value={value || ''}
        onChange={(e) => setValue(e.target.value)}
      />
    </div>
  )
}
```

**Critical Hooks**:
- `useField({ path })` - Get/set field value
- `useForm()` - Access form context
- `useFormFields()` - Read other field values
- `useTranslation()` - i18n support

### Example 2: JSON Field with Validation

```typescript
// src/components/admin/ValidatedJsonField/index.tsx
'use client'

import React from 'react'
import { useField } from '@payloadcms/ui'
import type { JSONFieldClientComponent } from 'payload'
import { z } from 'zod'

const MySchema = z.object({
  title: z.string().min(1),
  count: z.number().int().positive(),
})

export const ValidatedJsonField: JSONFieldClientComponent = ({ path, field }) => {
  const { value, setValue } = useField<unknown>({ path })
  const [jsonString, setJsonString] = React.useState(() =>
    JSON.stringify(value || {}, null, 2)
  )
  const [error, setError] = React.useState<string | null>(null)

  const handleChange = (newJson: string) => {
    setJsonString(newJson)
    setError(null)

    try {
      const parsed = JSON.parse(newJson)
      const validated = MySchema.parse(parsed)
      setValue(validated)
    } catch (e) {
      setError(e instanceof Error ? e.message : 'Invalid JSON')
    }
  }

  return (
    <div>
      <label>{field.label}</label>
      <textarea
        value={jsonString}
        onChange={(e) => handleChange(e.target.value)}
        rows={10}
      />
      {error && <div style={{ color: 'red' }}>{error}</div>}
    </div>
  )
}
```

### Example 3: Field with Dependencies

```typescript
'use client'
```

This project uses **vanilla CSS** for admin components (NOT Tailwind, NOT SCSS).

```typescript
// ✅ CORRECT: Import CSS file
import './index.css'

export const MyComponent = () => {
  return <div className="my-component">...</div>
}
```

```css
/* src/components/admin/MyComponent/index.css */

  padding: 1rem;

  margin-bottom: 0.5rem;
}
```

**Available CSS Variables**:
```css
/* Colors */
var(--theme-text)                  /* Main text color */
var(--theme-bg)                    /* Background */
var(--theme-elevation-0)           /* Lightest elevation */
var(--theme-elevation-300)         /* Border color */
var(--theme-elevation-500)         /* Muted elements */
var(--theme-success-500)           /* Success green */
var(--theme-error-500)             /* Error red */
var(--theme-warning-500)           /* Warning yellow */

/* Spacing */
var(--base)                        /* Base spacing unit */

var(--font-body)                   /* Body font */
var(--font-mono)                   /* Monospace font */
```

### ❌ DON'T: Use Tailwind in Admin Components

```typescript

  return <div className="p-4 bg-primary">...</div>  // Won't style!
}
```

**Why?** Admin panel and frontend have separate styling systems:
- **Frontend**: Tailwind CSS
- **Admin Panel**: Payload's CSS variables

## 🔍 Real Example: ExerciseContentEditor

This is the most complex custom component in the project.

### Features

1. **Block-based editor** - Add/edit/delete/reorder blocks
2. **Live preview** - See changes before saving
3. **JSON inspector** - Edit block JSON directly
4. **Media picker** - Attach images to blocks
5. **Block type selector** - Choose block type when adding
6. **Unsaved changes tracking** - Save button only when needed

### Key Implementation Details

```typescript
// src/components/admin/ExerciseContentEditor/index.tsx
'use client'

import React from 'react'
import { useField, useForm } from '@payloadcms/ui'
import './index.css'

export const ExerciseContentEditor: React.FC<{ path: string }> = ({ path }) => {
  const { value: fieldValue, setValue } = useField<any>({ path })
  const form = useForm()

  // Local state to prevent autosave
  const [localValue, setLocalValue] = React.useState(fieldValue)
  const [hasUnsavedChanges, setHasUnsavedChanges] = React.useState(false)

  // Update local value without triggering form.setModified
  const updateLocalValue = (newValue: any) => {
    setLocalValue(newValue)
    setHasUnsavedChanges(true)

    // Prevent autosave
    if (form.setModified) {
      setTimeout(() => form.setModified(false), 50)
    }
  }

  // Save changes manually
  const handleSave = async () => {
    setValue(localValue)

    // Mark form as modified to enable save button
    if (form.setModified) {
      form.setModified(true)
    }

    // Click the form's save button
    await new Promise(resolve => setTimeout(resolve, 150))
    const saveButton = document.querySelector('button[type="submit"]')
    if (saveButton) {
      saveButton.click()
    }

    setHasUnsavedChanges(false)
  }

  // Get blocks array safely
  const blocks = localValue?.blocks || []

  return (
    <div className="exercise-content-editor">
      <div className="editor-header">
        <h3>Exercise Content</h3>
        {hasUnsavedChanges && (
          <button onClick={handleSave} type="button">
            Save Changes
          </button>
        )}
      </div>
      <div className="block-list">
        {blocks.map((block, index) => (
          <BlockEditor
            key={block.id}
            block={block}
            onUpdate={(updates) => {
              const newBlocks = [...blocks]
              newBlocks[index] = { ...block, ...updates }
              updateLocalValue({ blocks: newBlocks })
            }}
          />
        ))}
      </div>
    </div>
  )
}
```

**Key Patterns**:
1. ✅ Use local state to prevent autosave
2. ✅ Provide manual save button
3. ✅ Safely access nested data (`value?.blocks || []`)
4. ✅ Use stable IDs for blocks (`block.id`)

## 🔍 Real Example: Coupons Admin Components

The Coupons collection (`src/server/payload/collections/Coupons.ts`) demonstrates several admin component patterns working together.

### Pattern 1: Cell Components for List View

Custom cells render computed/derived values in the list view. Each cell receives `cellData` and `fieldData` props from Payload:

```typescript
// src/ui/admin/Coupons/Cells/StatusCell/index.tsx
'use client'

import React from 'react'
import { useTranslation } from '@payloadcms/ui'
import { getCouponStrings } from '../../strings'

type CouponStatus = 'Active' | 'Expired' | 'Exhausted' | 'Inactive' | 'Scheduled'

const STATUS_CONFIG: Record<CouponStatus, { classes: string }> = {
  Active:    { classes: 'bg-success/15 text-success border border-success/30' },
  Expired:   { classes: 'bg-destructive/15 text-destructive border border-destructive/30' },
  Exhausted: { classes: 'bg-warning/15 text-warning border border-warning/30' },
  Inactive:  { classes: 'bg-muted text-muted-foreground border border-muted' },
  Scheduled: { classes: 'bg-primary/15 text-primary border border-primary/30' },
}

export const CouponStatusCell: React.FC<{ cellData?: string; fieldData?: CouponStatus }> = ({
  cellData, fieldData,
}) => {
  const { i18n } = useTranslation()
  const s = getCouponStrings(i18n.language)
  const status = (cellData || fieldData || 'Inactive') as CouponStatus
  const config = STATUS_CONFIG[status] ?? STATUS_CONFIG.Inactive

  const label = status === 'Active' ? s.statusActive
    : status === 'Expired' ? s.statusExpired
    : status === 'Exhausted' ? s.statusExhausted
    : status === 'Inactive' ? s.statusInactive
    : s.statusScheduled

  return (
    <span className={`inline-flex items-center gap-1 px-2 py-0.5 rounded-full text-label font-semibold ${config.classes}`}>
      {label}
    </span>
  )
}
```

Registration in collection config:

```typescript
// src/server/payload/collections/Coupons.ts
{
  name: 'status',
  type: 'text',
  admin: {
    components: {
      Cell: '@/ui/admin/Coupons/Cells/StatusCell#CouponStatusCell',
    },
  },
  hooks: {
    afterRead: [afterReadCouponStatus],
  },
},
```

### Pattern 2: UI Field Component (No Data Storage)

A `type: 'ui'` field renders a component without storing data — useful for display-only sections like progress bars:

```typescript
// src/ui/admin/Coupons/UsageProgressField/index.tsx
'use client'

import React from 'react'
import { useFormFields, useTranslation } from '@payloadcms/ui'
import { Progress } from '@/ui/web/components/progress'
import { getCouponStrings } from '../strings'

export const CouponUsageProgress: React.FC = () => {
  const { i18n } = useTranslation()
  const s = getCouponStrings(i18n.language)

  // Read sibling field values via useFormFields
  const usesCountField = useFormFields(([fields]) => fields.usesCount)
  const maxUsesField = useFormFields(([fields]) => fields.maxUses)

  const usesCount = (usesCountField?.value as number) ?? 0
  const maxUses = (maxUsesField?.value as number) ?? 0

  if (maxUses <= 0) return null  // Only show when maxUses > 0

  const percentage = Math.min(100, Math.round((usesCount / maxUses) * 100))
  const isExhausted = usesCount >= maxUses

  return (
    <div className="space-y-2 p-card-padding-sm border border-border rounded-lg bg-card">
      <div className="flex items-center justify-between">
        <span className="text-label font-semibold text-foreground">{s.usageProgress}</span>
        <span className={`text-label font-mono ${isExhausted ? 'text-destructive' : 'text-foreground'}`}>
          {usesCount} / {maxUses}
        </span>
      </div>
      <Progress value={percentage} className="h-2" />
    </div>
  )
}
```

Registration:

```typescript
{
  name: 'usageProgress',
  type: 'ui',
  admin: {
    components: {
      Field: '@/ui/admin/Coupons/UsageProgressField#CouponUsageProgress',
    },
  },
},
```

### Pattern 4: Computed Display Fields via afterRead Hooks

Virtual derived fields are computed at read time and only exist in the admin panel:

```typescript
// src/server/payload/hooks/coupons/computeListDisplayFields-hook.ts
export const afterReadCouponStatus: FieldAfterReadHook = ({ siblingData }) => {
  const { isActive, validFrom, validUntil, maxUses, usesCount } = siblingData

  if (!isActive) return 'Inactive'
  if (validFrom && new Date(validFrom) > new Date()) return 'Scheduled'
  if (validUntil && new Date(validUntil) < new Date()) return 'Expired'
  if (maxUses > 0 && usesCount >= maxUses) return 'Exhausted'
  return 'Active'
}
```

These fields are declared in the collection with `hooks.afterRead` but no database storage — they are computed on every read.

## 🎓 Common Patterns

### Pattern 1: Auto-fill from Another Field

```typescript
'use client'

import { useField, useFormFields } from '@payloadcms/ui'

export const SlugField: React.FC<{ path: string }> = ({ path }) => {
  const { value, setValue } = useField<string>({ path })
  const titleField = useFormFields(([fields]) => fields.title)

  React.useEffect(() => {
    if (!value && titleField?.value) {
      // Auto-generate slug from title
      const slug = titleField.value.toLowerCase().replace(/\s+/g, '-')
      setValue(slug)
    }
  }, [titleField?.value, value, setValue])

  return <input value={value || ''} onChange={(e) => setValue(e.target.value)} />
}
```

### Pattern 2: Load Example Data

```typescript
'use client'

import { useField } from '@payloadcms/ui'
import exampleData from './example.json'

export const TemplateField: React.FC<{ path: string }> = ({ path }) => {
  const { value, setValue } = useField<unknown>({ path })

  const handleLoadExample = () => {
    setValue(exampleData)
  }

  return (
    <div>
      <button type="button" onClick={handleLoadExample}>
        Load Example
      </button>
      <textarea
        value={JSON.stringify(value, null, 2)}
        onChange={(e) => {
          try {
            setValue(JSON.parse(e.target.value))
          } catch {}
        }}
      />
    </div>
  )
}
```

### Pattern 3: Prevent Form Autosave

```typescript
'use client'

import { useField, useForm } from '@payloadcms/ui'

export const ManualSaveField: React.FC<{ path: string }> = ({ path }) => {
  const { value, setValue } = useField<string>({ path })
  const form = useForm()
  const [localValue, setLocalValue] = React.useState(value)

  const handleChange = (newValue: string) => {
    setLocalValue(newValue)

    // Prevent autosave
    if (form.setModified) {
      setTimeout(() => form.setModified(false), 50)
    }
  }

  const handleSave = () => {
    setValue(localValue)
    if (form.setModified) {
      form.setModified(true)
    }
  }

  return (
    <div>
      <input value={localValue || ''} onChange={(e) => handleChange(e.target.value)} />
      <button type="button" onClick={handleSave}>Apply Changes</button>
    </div>
  )
}
```

## 🔒 Security Considerations

### Access Control in Components

Components run in the admin panel, which already requires authentication. However:

```typescript
'use client'

import { useAuth } from '@payloadcms/ui'

export const AdminOnlyField: React.FC<{ path: string }> = ({ path }) => {
  const { user } = useAuth()

  // Only show to admins
  if (user?.role !== 'admin') {
    return <div>Access denied</div>
  }

  return <div>Admin-only content</div>
}
```

### Validate User Input

```typescript
'use client'

import { useField } from '@payloadcms/ui'
import { z } from 'zod'

const urlSchema = z.string().url()

export const UrlField: React.FC<{ path: string }> = ({ path }) => {
  const { value, setValue } = useField<string>({ path })
  const [error, setError] = React.useState<string | null>(null)

  const handleChange = (newValue: string) => {
    try {
      urlSchema.parse(newValue)
      setValue(newValue)
      setError(null)
    } catch (e) {
      setError('Invalid URL')
    }
  }

  return (
    <div>
      <input value={value || ''} onChange={(e) => handleChange(e.target.value)} />
      {error && <span style={{ color: 'red' }}>{error}</span>}
    </div>
  )
}
```

## 🚫 Common Pitfalls

### ❌ Pitfall 1: Not Using 'use client'

```typescript
// ❌ WRONG: Missing 'use client'
import { useField } from '@payloadcms/ui'

export const MyField = ({ path }) => {
  const { value } = useField({ path })  // Error: hooks only work in client components
  return <div>{value}</div>
}

// ✅ CORRECT: Add 'use client'
'use client'
import { useField } from '@payloadcms/ui'

export const MyField = ({ path }) => {
  const { value } = useField({ path })
  return <div>{value}</div>
}
```

### ❌ Pitfall 2: Using Tailwind Classes

```typescript
// ❌ WRONG: Tailwind won't work
export const WrongStyling = () => {
  return <div className="p-4 bg-blue-500">Content</div>
}

// ✅ CORRECT: Use CSS variables
import './styles.css'

export const CorrectStyling = () => {
  return <div className="my-component">Content</div>
}
```


### ❌ Pitfall 3: Forgetting Import Map

```bash
# After creating new component:
pnpm run generate:importmap

# If you forget, you'll see:
# Error: Cannot find module '@/components/admin/MyNewField'
```

### ❌ Pitfall 4: Incorrect Path Syntax

```typescript
// ❌ WRONG: Relative path
components: {
  Field: './MyField',  // Won't work
}

// ❌ WRONG: Missing hash for named export
components: {
  Field: '@/components/admin/MyField',  // Looks for default export
}

// ✅ CORRECT: Absolute path with named export
components: {
  Field: '@/components/admin/MyField#MyField',
}
```

- [ ] **useField hook** used to get/set value
- [ ] **CSS file created** (if styling needed)
- [ ] **Registered in collection** with correct path syntax
- [ ] **Import map regenerated** (`pnpm run generate:importmap`)
